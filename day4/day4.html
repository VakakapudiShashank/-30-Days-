# Day 4 ‚Äî To-Do List App (CLI)
tasks = []

def show_tasks():
    if not tasks:
        print("üì≠ No tasks yet.")
    else:
        for i, task in enumerate(tasks, 1):
            print(f"{i}. {task}")

while True:
    print("\n1. View tasks\n2. Add task\n3. Remove task\n4. Mark task as done\n5. Clear all tasks\n6. Exit")
    choice = input("Enter choice: ")

    if choice == "1":
        show_tasks()
    elif choice == "2":
        task = input("Enter new task: ")
        tasks.append(task)
        print(f"‚úÖ '{task}' added.")
    elif choice == "3":
        show_tasks()
        idx = int(input("Enter task number: ")) - 1
        if 0 <= idx < len(tasks):
            removed = tasks.pop(idx)
            print(f"üóë '{removed}' removed.")
    elif choice == "4":
        show_tasks()
        idx = int(input("Enter task number: ")) - 1
        if 0 <= idx < len(tasks):
            print(f"üéØ Task {idx+1} marked as done.")
    elif choice == "5":
        confirm = input("Clear all tasks? (y/n): ")
        if confirm.lower() == "y":
            tasks.clear()
            print("üßπ All tasks cleared.")
    elif choice == "6":
        break
    else:
        print("‚ùå Invalid choice")
