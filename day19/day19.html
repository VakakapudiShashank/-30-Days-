# Day 19 ‚Äî Email & Phone Extractor
import re

def extract_contacts(text):
    emails = re.findall(r'[\\w\\.-]+@[\\w\\.-]+\\.\\w+', text)
    phones = re.findall(r'\\+?\\d[\\d\\s-]{7,}\\d', text)
    return emails, phones

if __name__ == "__main__":
    sample = '''
    Contact us at support@example.com or sales@myshop.org.
    Call +1-202-555-0187 or 9876543210 for more info.
    '''
    e, p = extract_contacts(sample)
    print("Emails:", e)
    print("Phones:", p)
