# Day 10: File Handling - Notes Saver
def save_note(filename, note):
    with open(filename, "a") as file:
        file.write(note + "\n")
    print("‚úÖ Note saved successfully!")

def read_notes(filename):
    try:
        with open(filename, "r") as file:
            content = file.readlines()
            if content:
                print("\nüìí Your Notes:")
                for idx, line in enumerate(content, start=1):
                    print(f"{idx}. {line.strip()}")
            else:
                print("üì≠ No notes found!")
    except FileNotFoundError:
        print("‚ùå No notes file found. Please add a note first.")

def main():
    filename = "notes.txt"
    while True:
        print("\n=== Notes Saver ===")
        print("1. Add Note")
        print("2. View Notes")
        print("3. Exit")
        choice = input("Enter choice: ")

        if choice == "1":
            note = input("Enter your note: ")
            save_note(filename, note)
        elif choice == "2":
            read_notes(filename)
        elif choice == "3":
            print("üëã Goodbye!")
            break
        else:
            print("‚ö† Invalid choice! Please try again.")

if __name__ == "__main__":
    main()
