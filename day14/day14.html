# Day 14 ‚Äî Student Grades Analyzer
from functools import reduce

students = {
    "Alice": 85,
    "Bob": 67,
    "Charlie": 90,
    "David": 45,
    "Eve": 72
}

# Increase all grades by 5% using map
scaled_grades = dict(map(lambda kv: (kv[0], int(kv[1] * 1.05)), students.items()))

# Filter passed students (>= 60)
passed = dict(filter(lambda kv: kv[1] >= 60, scaled_grades.items()))

# Compute average using reduce
avg_grade = reduce(lambda a, b: a + b, scaled_grades.values()) / len(scaled_grades)

# Find topper
topper = max(scaled_grades, key=scaled_grades.get)

print("Scaled Grades:", scaled_grades)
print("Passed Students:", passed)
print("Average Grade:", round(avg_grade, 2))
print("Topper:", topper)
